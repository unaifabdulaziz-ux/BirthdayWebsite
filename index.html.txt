<script>
  // Start Celebration
  function startCelebration() {
    document.getElementById('frontPage').style.display = 'none';
    document.getElementById('mainContent').style.display = 'block';
    revealLines();
  }

  // Sequential line reveal
  const lines = [
    "line1","line2","heart1","line3","line4","line5","line6",
    "line7","line8","line9","line10","line11","line12"
  ];

  function revealLines(){
    let i=0;
    function reveal(){
      if(i<lines.length){
        document.getElementById(lines[i]).style.display="block";
        i++;
        setTimeout(reveal,900); // 0.9 sec gap between lines
      } else {
        typeEnding();
      }
    }
    reveal();
  }

  // Typewriter ending
  const endingText = "Yeh sirf shuruaat hai… 💫 Main chahta hoon ke tumhari har birthday, har khushi, aur har khwab mein main tumhare saath hoon.";
  let j=0;
  function typeEnding(){
    const el = document.getElementById("endingText");
    el.style.display="block";
    function typeChar(){
      if(j<endingText.length){
        el.innerHTML += endingText.charAt(j);
        j++;
        setTimeout(typeChar,60);
      }
    }
    typeChar();
  }

  // Floating hearts with changing colors
  const colors = ["#ff2d55","#ff69b4","#ffd1dc","#ffffff","#c77dff"];
  setInterval(()=>{
    const heart=document.createElement("div");
    heart.className="bg-heart";
    heart.innerText="❤️";
    heart.style.left=Math.random()*100+"vw";
    heart.style.fontSize=(20+Math.random()*30)+"px";
    heart.style.color=colors[Math.floor(Math.random()*colors.length)];
    document.body.appendChild(heart);
    setTimeout(()=>heart.remove(),8000);
  },350);

</script>
